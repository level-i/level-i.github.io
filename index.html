ESTADO DEL SISTEMA (La memoria del Advisor) let leadData = { operacion: '', // Vender / Comprar inmueble: '', // Chalet / Piso / Villa zona: '', // La Ràpita / Alcanar telefono: '' // Contacto final }; let pasoActual = 0; // Controla en qué pregunta está el usuario // 2. CONFIGURACIÓN DE LA VOZ (Buscando el perfil profesional y elegante) function hablarAdvisor(texto) { const ssu = new SpeechSynthesisUtterance(texto); const voces = window.speechSynthesis.getVoices(); // Buscamos una voz femenina en español que suene natural (ej: Mónica o Google español hembra) let vozFemenina = voces.find(voz => voz.lang.includes('es') && (voz.name.includes('Google') || voz.name.includes('Monica') || voz.name.includes('Hazel'))); if (vozFemenina) ssu.voice = vozFemenina; ssu.rate = 0.95; // Un punto más pausada para que suene sugerente y profesional ssu.pitch = 1.05; // Ajuste de tono window.speechSynthesis.speak(ssu); } // 3. RECEPTOR DE VOZ (El oído del Advisor) const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)(); recognition.lang = 'es-ES'; recognition.interimResults = false; function escucharCliente() { document.getElementById('estado-ia').innerText = "ESCUCHANDO..."; recognition.start(); } recognition.onresult = function(event) { const respuesta = event.results[0][0].transcript.toLowerCase(); document.getElementById('estado-ia').innerText = "PROCESANDO..."; procesarFlujo(respuesta); }; recognition.onerror = function() { // Si falla la voz por silencio o ruido, el Advisor lo gestiona con elegancia hablarAdvisor("Disculpa, no te he entendido bien. ¿Podrías repetírmelo?"); document.getElementById('estado-ia').innerText = "IA ACTIVA"; }; // 4. EL CEREBRO DEL CUESTIONARIO (Árbol de decisión paso a paso) function iniciarCuestionario(tipo) { leadData.operacion = tipo; // 'Vender' pasoActual = 1; hablarAdvisor("Excelente elección. Para cruzar los datos con nuestra inteligencia artificial, ¿en qué localidad se encuentra tu propiedad? ¿En La Ràpita o en Alcanar?"); } function procesarFlujo(respuesta) { if (pasoActual === 1) { // Guardamos la zona if (respuesta.includes('rápita') || respuesta.includes('rapita')) leadData.zona = 'La Ràpita'; else if (respuesta.includes('alcanar')) leadData.zona = 'Alcanar'; else leadData.zona = respuesta; // Por si dice otra pasoActual = 2; hablarAdvisor("Entendido. ¿Y de qué tipo de inmueble se trata? ¿Es un piso, un chalet o una villa?"); return; } if (pasoActual === 2) { leadData.inmueble = respuesta; pasoActual = 3; hablarAdvisor("Perfecto. Por último, para registrar tu valoración en nuestra base de datos, por favor, dime tu número de teléfono en voz alta."); return; } if (pasoActual === 3) { // Limpiamos el texto por si dice números en letras leadData.telefono = respuesta; hablarAdvisor("Muchas gracias. Estoy transmitiendo los datos a la aplicación central de LEVEL de inmediato. Nos vemos muy pronto."); document.getElementById('estado-ia').innerText = "ENVIANDO DATOS..."; enviarDatosALark(); } } // 5. EL PUENTE CRÍTICO (Envío directo a tu App de empresa / Lark) function enviarDatosALark() { console.log("Inyectando datos en Lark...", leadData); // Aquí conectaremos tu Webhook real de Lark fetch('AQUÍ_PONDREMOS_TU_ENDPOINT_DE_LARK_O_APP', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(leadData) }) .then(res => { document.getElementById('estado-ia').innerText = "SISTEMA COMPLETADO"; console.log("¡Éxito! Fila creada en Lark."); }) .catch(err => console.error("Error de envío:", err)); }
